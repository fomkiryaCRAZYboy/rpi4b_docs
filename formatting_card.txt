sudo fdisk -l

# выбираем свою карту
sudo fdisk /dev/sdX


В fdisk:
o        # создаем новую DOS таблицу разделов
n        # новый раздел (primary)
[Enter]  # первый раздел
[Enter]  # начало раздела по умолчанию
+256M    # размер раздела (256 мегабайт)
t        # тип раздела
c        # W95 FAT32 (LBA)
w        # записать изменения

# Форматирование в FAT32:
sudo mkfs.vfat -F 32 -n BOOT /dev/sdX1
